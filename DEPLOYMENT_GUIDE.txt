STEP 8: CONTINUOUS MONITORING & LOG ROTATION

# Monitor Zeek flows

# Query database for recent activity
sqlite3 /opt/mod-router/mod-router.db << 'QUERY'
SELECT 
  d.hostname,
  COUNT(DISTINCT dns.queried_domain) as unique_domains,
  COUNT(dns.query_id) as total_queries,
  MAX(dns.timestamp) as last_activity
FROM devices d
LEFT JOIN dns_queries dns ON d.device_id = dns.device_id
WHERE dns.timestamp > datetime('now', '-1 hour')
GROUP BY d.device_id
ORDER BY total_queries DESC
LIMIT 10;
QUERY

# Compress logs (monthly rotation)
find /var/log/mod-router -name '*.log' -mtime +30 -exec gzip {} \;

# Backup database (weekly)
cp /opt/mod-router/mod-router.db /opt/mod-router/mod-router.db.$(date +%Y%m%d)
